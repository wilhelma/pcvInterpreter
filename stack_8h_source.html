33 template <typename Allocator>
34 class Stack {
35 public:
36  // Optimization note: Do not allocate memory for stack_ in constructor.
37  // Do it lazily when first Push() -> Expand() -> Resize().
38  Stack(Allocator* allocator, size_t stackCapacity) : allocator_(allocator), ownAllocator(0), stack_(0), stackTop_(0), stackEnd_(0), initialCapacity_(stackCapacity) {
39  RAPIDJSON_ASSERT(stackCapacity > 0);
40  if (!allocator_)
41  ownAllocator = allocator_ = new Allocator();
42  }
43 
44  ~Stack() {
45  Allocator::Free(stack_);
46  delete ownAllocator; // Only delete if it is owned by the stack
47  }

49  void Clear() { stackTop_ = stack_; }

51  void ShrinkToFit() {
52  if (Empty()) {
53  // If the stack is empty, completely deallocate the memory.
54  Allocator::Free(stack_);
55  stack_ = 0;
56  stackTop_ = 0;
57  stackEnd_ = 0;
58  }
59  else
60  Resize(GetSize());
61  }

63  // Optimization note: try to minimize the size of this function for force inline.
64  // Expansion is run very infrequently, so it is moved to another (probably non-inline) function.
65  template<typename T>
66  RAPIDJSON_FORCEINLINE T* Push(size_t count = 1) {
67  // Expand the stack if needed
68  if (stackTop_ + sizeof(T) * count >= stackEnd_)
69  Expand<T>(count);
70 
71  T* ret = reinterpret_cast<T*>(stackTop_);
72  stackTop_ += sizeof(T) * count;
73  return ret;
74  }

77  T* Pop(size_t count) {
78  RAPIDJSON_ASSERT(GetSize() >= count * sizeof(T));
79  stackTop_ -= count * sizeof(T);
80  return reinterpret_cast<T*>(stackTop_);
81  }

98  template<typename T>
99  void Expand(size_t count) {
100  // Only expand the capacity if the current stack exists. Otherwise just create a stack with initial capacity.
101  size_t newCapacity;
102  if (stack_ == 0)
103  newCapacity = initialCapacity_;
104  else {
105  newCapacity = GetCapacity();
106  newCapacity += (newCapacity + 1) / 2;
107  }
108  size_t newSize = GetSize() + sizeof(T) * count;
109  if (newCapacity < newSize)
110  newCapacity = newSize;
111 
112  Resize(newCapacity);
113  }
114 
115  void Resize(size_t newCapacity) {
116  const size_t size = GetSize(); // Backup the current size
117  stack_ = (char*)allocator_->Realloc(stack_, GetCapacity(), newCapacity);
118  stackTop_ = stack_ + size;
119  stackEnd_ = stack_ + newCapacity;
120  }
